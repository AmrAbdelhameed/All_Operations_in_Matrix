#include <iostream>
using namespace std;

int main()
{
    int a[100][100], b[100][100], mult[100][100], r1, c1, r2, c2, i, j, k;

    cout << "Enter rows and columns for first matrix : ";
    cin >> r1 >> c1;
    cout << "Enter rows and columns for second matrix : ";
    cin >> r2 >> c2;

    while (c1!=r2)
    {
        cout << "Error! column of first matrix not equal to row of second .";
        cout << endl;

        cout << "Enter rows and columns for first matrix : ";
        cin >> r1 >> c1;

        cout << "Enter rows and columns for second matrix : ";
        cin >> r2 >> c2;
    }
    cout << endl << "Enter elements of matrix 1 :" << endl;
    cout << endl;
    for(i = 0; i < r1; ++i)
        for(j = 0; j < c1; ++j)
        {
            cin >> a[i][j];
        }

    cout << endl << "Enter elements of matrix 2 :" << endl;
    cout << endl;
    for(i = 0; i < r2; ++i)
        for(j = 0; j < c2; ++j)
        {
            cin >> b[i][j];
        }

    // Initializing All = 0
    for(i = 0; i < r1; ++i)
        for(j = 0; j < c2; ++j)
        {
            mult[i][j]=0;
        }

    // Multiplying
    for(i = 0; i < r1; ++i)
        for(j = 0; j < c2; ++j)
            for(k = 0; k < c1; ++k)
            {
                mult[i][j] += a[i][k] * b[k][j];
            }

    //Display
    cout << endl << "Output New Matrix : " << endl;
    cout << endl;
    for(i = 0; i < r1; ++i)
        for(j = 0; j < c2; ++j)
        {
            cout <<mult[i][j]<<" ";
            if(j == c2-1)
                cout << endl << endl;
        }

    return 0;
}
